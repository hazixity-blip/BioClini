Convert CSV string data into a searchable database structure import csv matrix_records = {} f = io.StringIO(mock_database_csv.strip()) reader = csv.DictReader(f) for row in reader: matrix_records[row['drug']] = row st.set_page_config(page_title="Bio-Clinical Pipeline", page_icon="🧬", layout="centered") st.title("🧬 Bio-Clinical Informatics Platform") st.write("An API-driven system bridging molecular discovery data registries with clinical AMCI coding standards.") st.markdown("---") user_drug = st.text_input("Enter a Targeted Therapy Generic Drug Name:", placeholder="e.g., imatinib, paclitaxel, doxorubicin") if user_drug: clean_drug = user_drug.lower().strip() st.info(f"📡 Dispatching network request to official U.S. NIH PubChem Servers for: **{clean_drug.upper()}**") domain = "https://pubchem.ncbi.nlm.nih.gov" target_api_url = f"{domain}/rest/pug/compound/name/{clean_drug}/cids/JSON" try: response = requests.get(target_api_url, timeout=10) if response.status_code == 200: payload = response.json() compound_id = payload["IdentifierList"]["CID"] st.success(f"✅ Connection Established! Registry Entry Located.") formula_api_url = f"{domain}/rest/pug/compound/cid/{compound_id}/property/MolecularFormula/JSON" formula_response = requests.get(formula_api_url).json() formula_string = formula_response["PropertyTable"]["Properties"]["MolecularFormula"] col1, col2 = st.columns(2) with col1: st.metric(label="PubChem Compound ID", value=str(compound_id)) with col2: st.metric(label="Molecular Structure", value=str(formula_string)) # Query the separate tracking data sheet for matches if clean_drug in matrix_records: match_data = matrix_records[clean_drug] st.markdown("### 📋 AMCI Standardized Alphabetic Index Pathway") st.code(f""" Index Search Path: └── 1. Main Term: {match_data['main_term']} └── 2. Subterm: {match_data['subterm_1']} └── 3. Specificity: {match_data['subterm_2']} └── 🎯 Final Tabular Verification: {match_data['icd10']} """) st.markdown("### 🚨 Coding Guideline & Rationale") st.warning(match_data['guideline']) st.markdown("### 🏥 Clinical Data Summary") st.markdown(f""" * **🧬 Biomarker Target:** {match_data['target']} * **损 Linked Indication:** {match_data['disease']} """) else: st.warning("⚠️ Compound validated on federal network, but administrative ICD-10 clinical data translation requires database mapping expansion.") else: st.error("❌ Compound query rejected: Molecular structure name not recognized by NIH registries.") except Exception as network_issue: st.error(f"❌ Network handshaking error failed: {network_issue}")
